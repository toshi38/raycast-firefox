try_node() {
  local node_path="$1"
  local label="$2"
  if [ -x "$node_path" ] && check_node_version "$node_path"; then
    local version
    version="$("$node_path" --version 2>/dev/null)"
    log_info "Using $label: $node_path ($version)"
    exec "$node_path" "$BUNDLE"
  fi
}
